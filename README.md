chmod +x setup.sh
./setup.sh

# Install packages for custom zsh config
chmod +x zsh-setup.sh
./zsh-setup.sh

cp .zshrc ~/.zshrc

source ~/.zshrc

# Setup the powerlevel 10k zsh theme
p10k configure
